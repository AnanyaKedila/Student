CREATE TABLE STUDENT(
	USN char(4),
	Sname varchar(25) NOT NULL,
	DOB date,
	Branch char(5),
	Mark1 int(3),
	Mark2 int(3),
	Mark3 int(3),
	PRIMARY KEY (USN)
);
